G.1.2　函数：__vmalloc

源码位置：mm/vmalloc.c

本函数有三项任务：将请求大小按页对齐；请求 get_vm_area() 为该请求找到一块区域；并使用 vmalloc_area_pages() 为这些页分配 PTE。

277     }

G.1.3　函数：get_vm_area

源码位置：mm/vmalloc.c

为分配一个 vm_struct 所需的区域，会请求 slab 分配器通过 kmalloc() 提供所需内存。然后线性遍历 vm_struct 链表，寻找一块足够大的区域以满足请求，且在区域末尾包含一页填充。

                                   unsigned long flags)

201     if (!area)

208     }

211     write_lock(&vmlist_lock);
212     for (p = &vmlist; (tmp = *p) ; p = &tmp->next) {
213         if ((size + addr) < addr)
214             goto out;
215         if (size + addr <= (unsigned long) tmp->addr)

G.1.5　函数：__vmalloc_area_pages

源码位置：mm/vmalloc.c

这是标准页表遍历函数的开始。该顶层函数会遍历地址范围内的所有 PGD。对每个 PGD，它会调用 pmd_alloc() 分配一个 PMD 目录，然后对该目录调用 alloc_area_pmd()。

155 static inline int __vmalloc_area_pages (unsigned long address,
156                                         unsigned long size,
157                                         int gfp_mask,
158                                         pgprot_t prot,
159                                         struct page ***pages)
160 {
161     pgd_t * dir;
162     unsigned long end = address + size;
163     int ret;
164 
165     dir = pgd_offset_k(address);
166     spin_lock(&init_mm.page_table_lock);
167     do {
168         pmd_t *pmd;
169         
170         pmd = pmd_alloc(&init_mm, dir, address);
171         ret = -ENOMEM;
172         if (!pmd)
173             break;
174 
175         ret = -ENOMEM;
176         if (alloc_area_pmd(pmd, address, end - address, 
                       gfp_mask, prot, pages))
177             break;
178 
179         address = (address + PGDIR_SIZE) & PGDIR_MASK;

181 
182         ret = 0;

185     flush_cache_all();

187 }
